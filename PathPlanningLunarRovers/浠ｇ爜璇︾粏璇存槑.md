### 2.2 d3qn_network.py - 神经网络模块

#### 网络架构

```python
class DuelingDQN(nn.Module):
    def __init__(self):
        # 卷积层：处理深度图
        self.conv1 = nn.Conv2d(4, 32, kernel_size=8, stride=4)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=4, stride=2)
        self.conv3 = nn.Conv2d(64, 64, kernel_size=3, stride=1)

        # 全连接层：融合特征
        self.fc_cnn = nn.Linear(cnn_output_size, 256)
        self.fc_add = nn.Linear(8, 64)  # 附加特征
        self.fc_combined = nn.Linear(320, 512)

        # Dueling架构
        self.value_stream = nn.Linear(512, 1)      # V(s)
        self.advantage_stream = nn.Linear(512, 9)  # A(s,a)
```

**前向传播：**
```python
def forward(self, depth_image, additional_features):
    # Step 1: CNN特征提取
    x = F.relu(self.conv1(depth_image))
    x = F.relu(self.conv2(x))
    x = F.relu(self.conv3(x))
    x = x.view(batch_size, -1)

    # Step 2: 多模态融合
    cnn_feat = F.relu(self.fc_cnn(x))
    add_feat = F.relu(self.fc_add(additional_features))
    combined = torch.cat([cnn_feat, add_feat], dim=1)

    # Step 3: Dueling合并
    features = F.relu(self.fc_combined(combined))
    value = self.value_stream(features)
    advantage = self.advantage_stream(features)

    # Q(s,a) = V(s) + (A(s,a) - mean(A(s,:)))
    q_values = value + (advantage - advantage.mean(dim=1, keepdim=True))

    return q_values
```

**Dueling架构优势：**
- 分离状态价值和动作优势
- 在某些状态下，动作选择不重要时，仍能准确评估状态价值
- 提高学习效率

#### 核心方法详解

##### select_action() - 动作选择

```python
def select_action(self, state, additional_features, training=True):
    """
    ε-greedy策略选择动作

    Args:
        state: 深度图 (4, 80, 64)
        additional_features: 附加特征 (8,)
        training: 是否训练模式

    Returns:
        action: int (0-8)
    """
    if training and random.random() < self.epsilon:
        # 探索：随机动作
        return random.randrange(self.action_dim)
    else:
        # 利用：贪心动作
        with torch.no_grad():
            state_t = torch.FloatTensor(state).unsqueeze(0)
            features_t = torch.FloatTensor(additional_features).unsqueeze(0)
            q_values = self.policy_net(state_t, features_t)
            return q_values.argmax(dim=1).item()
```

##### train() - 训练网络

```python
def train(self):
    """
    从回放缓冲区采样并更新网络
    """
    if len(self.replay_buffer) < config.MIN_REPLAY_SIZE:
        return None  # 样本不足，跳过训练

    # Step 1: 采样经验（带优先级）
    if self.use_per:
        batch, indices, weights = self.replay_buffer.sample(config.BATCH_SIZE)
    else:
        batch = self.replay_buffer.sample(config.BATCH_SIZE)
        weights = torch.ones(config.BATCH_SIZE)

    # Step 2: 解包批次数据
    states, actions, rewards, next_states, dones, features, next_features = batch

    # Step 3: 计算当前Q值
    current_q = self.policy_net(states, features).gather(1, actions)

    # Step 4: 计算目标Q值（Double DQN）
    with torch.no_grad():
        # 用policy网络选择动作
        next_actions = self.policy_net(next_states, next_features).argmax(1, keepdim=True)
        # 用target网络评估Q值
        next_q = self.target_net(next_states, next_features).gather(1, next_actions)
        target_q = rewards + (1 - dones) * config.GAMMA * next_q

    # Step 5: 计算损失
    td_errors = current_q - target_q
    loss = (weights * td_errors.pow(2)).mean()

    # Step 6: 反向传播
    self.optimizer.zero_grad()
    loss.backward()
    torch.nn.utils.clip_grad_norm_(self.policy_net.parameters(), config.GRAD_CLIP_NORM)
    self.optimizer.step()

    # Step 7: 更新优先级
    if self.use_per:
        priorities = td_errors.abs().detach().cpu().numpy() + config.PER_EPSILON
        self.replay_buffer.update_priorities(indices, priorities)

    # Step 8: 软更新目标网络
    self.soft_update_target_network()

    return loss.item()
```

**Double DQN原理：**
- 问题：标准DQN容易高估Q值
- 解决：用policy网络选动作，target网络评估Q值
- 减少过估计偏差，提高稳定性

### 2.4 astar.py - A*路径规划模块

#### 栅格地图创建

```python
def create_grid_map(width, height, resolution, obstacles):
    """
    创建栅格地图并进行障碍物膨胀

    Args:
        width: 环境宽度 (m)
        height: 环境高度 (m)
        resolution: 栅格分辨率 (m/grid)
        obstacles: 障碍物列表 [(x, y, radius), ...]

    Returns:
        grid_map: 2D numpy数组，0=可通行，1=障碍物
    """
    grid_width = int(width / resolution)
    grid_height = int(height / resolution)
    grid_map = np.zeros((grid_height, grid_width), dtype=np.uint8)

    # 障碍物膨胀
    inflation_radius = (config.OBSTACLE_RADIUS + config.ROVER_SAFE_RADIUS) / resolution

    for obs_x, obs_y, obs_r in obstacles:
        grid_x = int(obs_x / resolution)
        grid_y = int(obs_y / resolution)

        # 在栅格地图上标记障碍物
        for i in range(grid_height):
            for j in range(grid_width):
                dist = np.sqrt((i - grid_y)**2 + (j - grid_x)**2)
                if dist <= inflation_radius:
                    grid_map[i, j] = 1

    return grid_map
```

**膨胀策略：**
- 膨胀半径 = 障碍物半径 + 月球车安全半径
- 确保规划的路径与障碍物保持安全距离

#### A*算法实现

```python
class AStar:
    def plan(self, start, goal):
        """
        A*路径规划

        Args:
            start: (x, y) 起点坐标(米)
            goal: (x, y) 终点坐标(米)

        Returns:
            path: 路径点列表 [(x1,y1), (x2,y2), ...]
        """
        # 转换为栅格坐标
        start_grid = self.world_to_grid(start)
        goal_grid = self.world_to_grid(goal)

        # 初始化开放列表和关闭列表
        open_set = []
        heapq.heappush(open_set, (0, start_grid))
        came_from = {}

        g_score = {start_grid: 0}
        f_score = {start_grid: self.heuristic(start_grid, goal_grid)}

        while open_set:
            current = heapq.heappop(open_set)[1]

            if current == goal_grid:
                # 找到路径，回溯
                return self.reconstruct_path(came_from, current)

            for neighbor in self.get_neighbors(current):
                if self.grid_map[neighbor[1], neighbor[0]] == 1:
                    continue  # 跳过障碍物

                tentative_g = g_score[current] + self.cost(current, neighbor)

                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor, goal_grid)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))

        return None  # 未找到路径
```

**启发式函数：**
```python
def heuristic(self, node, goal):
    if config.ASTAR_HEURISTIC == 'manhattan':
        return abs(node[0] - goal[0]) + abs(node[1] - goal[1])
    else:  # euclidean
        return np.sqrt((node[0] - goal[0])**2 + (node[1] - goal[1])**2)
```

### 2.5 prioritized_replay.py - 优先经验回放

#### SumTree数据结构

```python
class SumTree:
    """
    二叉堆结构，支持O(log n)采样

    结构:
        优先级树: [总和, 左子树和, 右子树和, ..., 叶子节点优先级]
        数据数组: [experience1, experience2, ...]
    """

    def __init__(self, capacity):
        self.capacity = capacity
        self.tree = np.zeros(2 * capacity - 1)  # 完全二叉树
        self.data = np.zeros(capacity, dtype=object)
        self.write_idx = 0
        self.n_entries = 0
```

**添加经验：**
```python
def add(self, priority, data):
    """
    添加经验及其优先级
    """
    tree_idx = self.write_idx + self.capacity - 1
    self.data[self.write_idx] = data
    self.update(tree_idx, priority)

    self.write_idx = (self.write_idx + 1) % self.capacity
    if self.n_entries < self.capacity:
        self.n_entries += 1
```

**采样算法：**
```python
def sample(self, batch_size):
    """
    按优先级采样

    流程:
    1. 将总优先级分成batch_size个区间
    2. 在每个区间内随机采样
    3. 确保采样的多样性
    """
    batch = []
    indices = []
    priorities = []

    segment = self.tree[0] / batch_size  # 总优先级/批大小

    for i in range(batch_size):
        a = segment * i
        b = segment * (i + 1)
        value = np.random.uniform(a, b)

        idx, priority, data = self.get(value)
        batch.append(data)
        indices.append(idx)
        priorities.append(priority)

    return batch, indices, priorities
```

**重要性采样权重：**
```python
def calculate_weights(self, priorities, beta):
    """
    计算重要性采样权重，校正偏差

    weight_i = (N * P(i))^(-β) / max_weight
    """
    probs = priorities / self.tree[0]  # 归一化概率
    weights = (self.n_entries * probs) ** (-beta)
    weights /= weights.max()  # 归一化到[0,1]
    return weights
```

## 3. 训练流程详解

### 3.1 完整训练循环

```python
def train(args):
    # ========== 初始化 ==========
    set_random_seed(args.seed)
    env = LunarEnvironment(stage=args.stage, render=args.render)
    agent = create_agent(args.method)
    logger = Logger(args.save_dir)

    # ========== 训练循环 ==========
    for episode in range(args.episodes):
        # 1. 重置环境
        state = env.reset()
        additional_features = env.get_additional_features()

        # 2. 全局路径规划（如果使用A*）
        if agent.use_astar:
            start = (env.rover.x, env.rover.y)
            goal = (env.target_x, env.target_y)
            global_path = agent.plan_global_path(start, goal)

        # 3. 回合循环
        episode_reward = 0
        step = 0
        done = False

        while not done and step < config.MAX_STEPS_PER_EPISODE:
            # 3.1 选择动作
            action = agent.select_action(state, additional_features, training=True)

            # 3.2 执行动作
            next_state, reward, done, info = env.step(action)
            next_features = env.get_additional_features()

            # 3.3 存储经验
            agent.store_experience(
                state, action, reward, next_state, done,
                additional_features, next_features
            )

            # 3.4 训练网络
            loss = agent.train()

            # 3.5 更新状态
            state = next_state
            additional_features = next_features
            episode_reward += reward
            step += 1

            # 3.6 渲染（可选）
            if args.render:
                env.render()

        # 4. 更新超参数
        agent.update_epsilon()
        agent.update_learning_rate()

        # 5. 记录日志
        logger.log(episode, episode_reward, info)

        # 6. 保存模型
        if episode % config.SAVE_MODEL_FREQ == 0:
            agent.save_model(f"{args.save_dir}/model_ep{episode}.pth")

    # ========== 训练结束 ==========
    agent.save_model(f"{args.save_dir}/final_model.pth")
    logger.plot_results()
```

### 3.2 超参数衰减策略

#### 探索率衰减
```python
def update_epsilon(self):
    """
    指数衰减探索率

    ε_t = max(ε_min, ε_start * decay^t)
    """
    self.epsilon = max(
        config.EPSILON_MIN,
        self.epsilon * config.EPSILON_DECAY
    )
```

**衰减曲线：**
- 起始: ε = 1.0 (100%探索)
- 约1150回合: ε ≈ 0.10 (10%探索)
- 保持最小探索率，避免陷入局部最优

#### 学习率衰减
```python
def update_learning_rate(self):
    """
    学习率衰减
    """
    new_lr = max(
        config.LR_MIN,
        self.lr * config.LR_DECAY_RATE
    )
    for param_group in self.optimizer.param_groups:
        param_group['lr'] = new_lr
    self.lr = new_lr
```

```python
# 在perception.py中添加
class LiDARSensor:
    def __init__(self, num_beams=360, max_range=5.0):
        self.num_beams = num_beams
        self.max_range = max_range

    def get_scan(self, rover_pos, rover_theta, obstacles):
        """
        获取激光雷达扫描数据

        Returns:
            scan: numpy数组 (num_beams,) - 每个方向的距离
        """
        scan = np.ones(self.num_beams) * self.max_range
        angles = np.linspace(0, 2*np.pi, self.num_beams)

        for i, angle in enumerate(angles):
            global_angle = rover_theta + angle
            direction = np.array([
                np.cos(global_angle),
                np.sin(global_angle)
            ])

            # 光线追踪找最近障碍物
            min_dist = self.max_range
            for obs in obstacles:
                dist = self.ray_circle_intersection(
                    rover_pos, direction, obs.position, obs.radius
                )
                if dist is not None and dist < min_dist:
                    min_dist = dist

            scan[i] = min_dist

        return scan
```

### 7.2 实现新的规划算法

**示例：RRT（快速随机树）算法**

```python
# 创建rrt.py文件
class RRT:
    def __init__(self, grid_map, resolution):
        self.grid_map = grid_map
        self.resolution = resolution
        self.max_iterations = 5000
        self.step_size = 0.5  # 米

    def plan(self, start, goal):
        """
        RRT路径规划
        """
        tree = {start: None}

        for _ in range(self.max_iterations):
            # 1. 随机采样点
            random_point = self.random_sample(goal)

            # 2. 找最近节点
            nearest_node = self.find_nearest(tree, random_point)

            # 3. 扩展树
            new_node = self.steer(nearest_node, random_point)

            # 4. 碰撞检测
            if self.is_collision_free(nearest_node, new_node):
                tree[new_node] = nearest_node

                # 5. 检查是否到达目标
                if self.distance(new_node, goal) < self.step_size:
                    tree[goal] = new_node
                    return self.extract_path(tree, goal)

        return None  # 未找到路径
```
